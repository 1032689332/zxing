396   public static void corrupt(int[] received, int howMany, Random random, int max) {
397     BitSet corrupted = new BitSet(received.length);
398     for (int j = 0; j < howMany; j++) {
399       int location = random.nextInt(received.length);
400       int value = random.nextInt(max);
401       if (corrupted.get(location) || received[location] == value) {
402         j--;
403       } else {
404         corrupted.set(location);
405         received[location] = value;
406       }
407     }
408   }

410   private static void testEncodeDecodeRandom(GenericGF field, int dataSize, int ecSize) {
411     assertTrue("Invalid data size for " + field, dataSize > 0 && dataSize <= field.getSize() - 3);
412     assertTrue("Invalid ECC size for " + field, ecSize > 0 && ecSize + dataSize <= field.getSize());
413     ReedSolomonEncoder encoder = new ReedSolomonEncoder(field);
414     int[] message = new int[dataSize + ecSize];
415     int[] dataWords = new int[dataSize];
416     int[] ecWords = new int[ecSize];
417     Random random = getPseudoRandom();
418     int iterations = field.getSize() > 256 ? 1 : DECODER_RANDOM_TEST_ITERATIONS;
419     for (int i = 0; i < iterations; i++) {
420       // generate random data
421       for (int k = 0; k < dataSize; k++) {
422         dataWords[k] = random.nextInt(field.getSize());
423       }
424       // generate ECC words
425       System.arraycopy(dataWords, 0, message, 0, dataWords.length);
426       encoder.encode(message, ecWords.length);
427       System.arraycopy(message, dataSize, ecWords, 0, ecSize);
428       // check to see if Decoder can fix up to ecWords/2 random errors
429       testDecoder(field, dataWords, ecWords);
430     }
431   }

438   private static void testEncoder(GenericGF field, int[] dataWords, int[] ecWords) {
439     ReedSolomonEncoder encoder = new ReedSolomonEncoder(field);
440     int[] messageExpected = new int[dataWords.length + ecWords.length];
441     int[] message = new int[dataWords.length + ecWords.length];
442     System.arraycopy(dataWords, 0, messageExpected, 0, dataWords.length);
443     System.arraycopy(ecWords, 0, messageExpected, dataWords.length, ecWords.length);
444     System.arraycopy(dataWords, 0, message, 0, dataWords.length);
445     encoder.encode(message, ecWords.length);
446     assertDataEquals("Encode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed", 
447                      messageExpected, message);
448   }

450   private static void testDecoder(GenericGF field, int[] dataWords, int[] ecWords) {
451     ReedSolomonDecoder decoder = new ReedSolomonDecoder(field);
452     int[] message = new int[dataWords.length + ecWords.length];
453     int maxErrors = ecWords.length / 2;
454     Random random = getPseudoRandom();
455     int iterations = field.getSize() > 256 ? 1 : DECODER_TEST_ITERATIONS;
456     for (int j = 0; j < iterations; j++) {
457       for (int i = 0; i < ecWords.length; i++) {
458         if (i > 10 && i < ecWords.length / 2 - 10) {
459           // performance improvement - skip intermediate cases in long-running tests 
460           i += ecWords.length / 10;
461         }
462         System.arraycopy(dataWords, 0, message, 0, dataWords.length);
463         System.arraycopy(ecWords, 0, message, dataWords.length, ecWords.length);
464         corrupt(message, i, random, field.getSize());
465         try {
466           decoder.decode(message, ecWords.length);
467         } catch (ReedSolomonException e) {
468           // fail only if maxErrors exceeded
469           assertTrue("Decode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed at " + 
470                          i + " errors: " + e,
471                      i > maxErrors);
472           // else stop
473           break;
474         }
475         if (i < maxErrors) {
476           assertDataEquals("Decode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed at " + 
477                            i + " errors",
478                            dataWords, 
479                            message);
480         }
481       }
482     }
483   }

485   private static void assertDataEquals(String message, int[] expected, int[] received) {
486     for (int i = 0; i < expected.length; i++) {
487       if (expected[i] != received[i]) {
488         fail(message + ". Mismatch at " + i + ". Expected " + arrayToString(expected) + ", got " + 
489              arrayToString(Arrays.copyOf(received, expected.length)));
490       }
491     }
492   }

497       sb.append(String.format(i > 0 ? ",%X" : "%X", data[i]));
498     }
499     return sb.append('}').toString();
