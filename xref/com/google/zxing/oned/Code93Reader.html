31  /**
32   * <p>Decodes Code 93 barcodes.</p>
33   *
34   * @author Sean Owen
35   * @see Code39Reader
36   */
37  public final class Code93Reader extends OneDReader {
38  
39    // Note that 'abcd' are dummy characters in place of control characters.
40    private static final String ALPHABET_STRING = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ-. $/+%abcd*";
41    private static final char[] ALPHABET = ALPHABET_STRING.toCharArray();
42  
43    /**
44     * These represent the encodings of characters, as patterns of wide and narrow bars.
45     * The 9 least-significant bits of each int correspond to the pattern of wide and narrow.
46     */
47    private static final int[] CHARACTER_ENCODINGS = {
48        0x114, 0x148, 0x144, 0x142, 0x128, 0x124, 0x122, 0x150, 0x112, 0x10A, // 0-9
49        0x1A8, 0x1A4, 0x1A2, 0x194, 0x192, 0x18A, 0x168, 0x164, 0x162, 0x134, // A-J
50        0x11A, 0x158, 0x14C, 0x146, 0x12C, 0x116, 0x1B4, 0x1B2, 0x1AC, 0x1A6, // K-T
51        0x196, 0x19A, 0x16C, 0x166, 0x136, 0x13A, // U-Z
52        0x12E, 0x1D4, 0x1D2, 0x1CA, 0x16E, 0x176, 0x1AE, // - - %
53        0x126, 0x1DA, 0x1D6, 0x132, 0x15E, // Control chars? $-*
54    };
55    private static final int ASTERISK_ENCODING = CHARACTER_ENCODINGS[47];
56  
57    private final StringBuilder decodeRowResult;
58    private final int[] counters;
59  
60    public Code93Reader() {
61      decodeRowResult = new StringBuilder(20);
62      counters = new int[6];
63    }
64  
65    @Override
66    public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints)
67        throws NotFoundException, ChecksumException, FormatException {
68  
69      int[] start = findAsteriskPattern(row);
70      // Read off white space    
71      int nextStart = row.getNextSet(start[1]);
72      int end = row.getSize();
73  
74      int[] theCounters = counters;
75      Arrays.fill(theCounters, 0);
76      StringBuilder result = decodeRowResult;
77      result.setLength(0);
78  
79      char decodedChar;
80      int lastStart;
81      do {
82        recordPattern(row, nextStart, theCounters);
83        int pattern = toPattern(theCounters);
84        if (pattern < 0) {
85          throw NotFoundException.getNotFoundInstance();
86        }
87        decodedChar = patternToChar(pattern);
88        result.append(decodedChar);
89        lastStart = nextStart;
90        for (int counter : theCounters) {
91          nextStart += counter;
92        }
93        // Read off white space
94        nextStart = row.getNextSet(nextStart);
95      } while (decodedChar != '*');
96      result.deleteCharAt(result.length() - 1); // remove asterisk
97  
98      int lastPatternSize = 0;
99      for (int counter : theCounters) {
100       lastPatternSize += counter;
101     }
102 
103     // Should be at least one more black module
104     if (nextStart == end || !row.get(nextStart)) {
105       throw NotFoundException.getNotFoundInstance();
106     }
107 
108     if (result.length() < 2) {
109       // false positive -- need at least 2 checksum digits
110       throw NotFoundException.getNotFoundInstance();
111     }
112 
113     checkChecksums(result);
114     // Remove checksum digits
115     result.setLength(result.length() - 2);
116 
117     String resultString = decodeExtended(result);
118 
119     float left = (float) (start[1] + start[0]) / 2.0f;
120     float right = lastStart + lastPatternSize / 2.0f;
121     return new Result(
122         resultString,
123         null,
124         new ResultPoint[]{
125             new ResultPoint(left, (float) rowNumber),
126             new ResultPoint(right, (float) rowNumber)},
127         BarcodeFormat.CODE_93);
128 
129   }

131   private int[] findAsteriskPattern(BitArray row) throws NotFoundException {
132     int width = row.getSize();
133     int rowOffset = row.getNextSet(0);
134 
135     Arrays.fill(counters, 0);
136     int[] theCounters = counters;
137     int patternStart = rowOffset;
138     boolean isWhite = false;
139     int patternLength = theCounters.length;
140 
141     int counterPosition = 0;
142     for (int i = rowOffset; i < width; i++) {
143       if (row.get(i) ^ isWhite) {
144         theCounters[counterPosition]++;
145       } else {
146         if (counterPosition == patternLength - 1) {
147           if (toPattern(theCounters) == ASTERISK_ENCODING) {
148             return new int[]{patternStart, i};
149           }
150           patternStart += theCounters[0] + theCounters[1];
151           System.arraycopy(theCounters, 2, theCounters, 0, patternLength - 2);
152           theCounters[patternLength - 2] = 0;
153           theCounters[patternLength - 1] = 0;
154           counterPosition--;
155         } else {
156           counterPosition++;
157         }
158         theCounters[counterPosition] = 1;
159         isWhite = !isWhite;
160       }
161     }
162     throw NotFoundException.getNotFoundInstance();
163   }
164 
165   private static int toPattern(int[] counters) {
166     int max = counters.length;
167     int sum = 0;
168     for (int counter : counters) {
169       sum += counter;
170     }
171     int pattern = 0;
172     for (int i = 0; i < max; i++) {
173       int scaled = Math.round(counters[i] * 9.0f / sum);
174       if (scaled < 1 || scaled > 4) {
175         return -1;
176       }
177       if ((i & 0x01) == 0) {
178         for (int j = 0; j < scaled; j++) {
179           pattern = (pattern << 1) | 0x01;
180         }
181       } else {
182         pattern <<= scaled;
183       }
184     }
185     return pattern;
186   }

188   private static char patternToChar(int pattern) throws NotFoundException {
189     for (int i = 0; i < CHARACTER_ENCODINGS.length; i++) {
190       if (CHARACTER_ENCODINGS[i] == pattern) {
191         return ALPHABET[i];
192       }
193     }
194     throw NotFoundException.getNotFoundInstance();
195   }

197   private static String decodeExtended(CharSequence encoded) throws FormatException {
198     int length = encoded.length();
199     StringBuilder decoded = new StringBuilder(length);
200     for (int i = 0; i < length; i++) {
201       char c = encoded.charAt(i);
202       if (c >= 'a' && c <= 'd') {
203         if (i >= length - 1) {
204           throw FormatException.getFormatInstance();
205         }
206         char next = encoded.charAt(i + 1);
207         char decodedChar = '\0';
208         switch (c) {
209           case 'd':
210             // +A to +Z map to a to z
211             if (next >= 'A' && next <= 'Z') {
212               decodedChar = (char) (next + 32);
213             } else {
214               throw FormatException.getFormatInstance();
215             }
216             break;
217           case 'a':
218             // $A to $Z map to control codes SH to SB
219             if (next >= 'A' && next <= 'Z') {
220               decodedChar = (char) (next - 64);
221             } else {
222               throw FormatException.getFormatInstance();
223             }
224             break;
225           case 'b':
226             if (next >= 'A' && next <= 'E') {
227               // %A to %E map to control codes ESC to USep
228               decodedChar = (char) (next - 38);
229             } else if (next >= 'F' && next <= 'J') {
230               // %F to %J map to ; < = > ?
231               decodedChar = (char) (next - 11);
232             } else if (next >= 'K' && next <= 'O') {
233               // %K to %O map to [ \ ] ^ _
234               decodedChar = (char) (next + 16);
235             } else if (next >= 'P' && next <= 'S') {
236               // %P to %S map to { | } ~
237               decodedChar = (char) (next + 43);
238             } else if (next >= 'T' && next <= 'Z') {
239               // %T to %Z all map to DEL (127)
240               decodedChar = 127;
241             } else {
242               throw FormatException.getFormatInstance();
243             }
244             break;
245           case 'c':
246             // /A to /O map to ! to , and /Z maps to :
247             if (next >= 'A' && next <= 'O') {
248               decodedChar = (char) (next - 32);
249             } else if (next == 'Z') {
250               decodedChar = ':';
251             } else {
252               throw FormatException.getFormatInstance();
253             }
254             break;
255         }
256         decoded.append(decodedChar);
257         // bump up i again since we read two characters
258         i++;
259       } else {
260         decoded.append(c);
261       }
262     }
263     return decoded.toString();
264   }

266   private static void checkChecksums(CharSequence result) throws ChecksumException {
267     int length = result.length();
268     checkOneChecksum(result, length - 2, 20);
269     checkOneChecksum(result, length - 1, 15);
270   }
271 
272   private static void checkOneChecksum(CharSequence result, int checkPosition, int weightMax)
273       throws ChecksumException {
274     int weight = 1;
275     int total = 0;
276     for (int i = checkPosition - 1; i >= 0; i--) {
277       total += weight * ALPHABET_STRING.indexOf(result.charAt(i));
278       if (++weight > weightMax) {
279         weight = 1;
280       }
281     }
282     if (result.charAt(checkPosition) != ALPHABET[total % 47]) {
283       throw ChecksumException.getChecksumInstance();
284     }
285   }
