134   static {
135     //Construct inverse lookups
136     Arrays.fill(MIXED, (byte) -1);
137     for (byte i = 0; i < TEXT_MIXED_RAW.length; i++) {
138       byte b = TEXT_MIXED_RAW[i];
139       if (b > 0) {
140         MIXED[b] = i;
141       }
142     }
143     Arrays.fill(PUNCTUATION, (byte) -1);
144     for (byte i = 0; i < TEXT_PUNCTUATION_RAW.length; i++) {
145       byte b = TEXT_PUNCTUATION_RAW[i];
146       if (b > 0) {
147         PUNCTUATION[b] = i;
148       }
149     }
150   }

152   /**
153    * Performs high-level encoding of a PDF417 message using the algorithm described in annex P
154    * of ISO/IEC 15438:2001(E). If byte compaction has been selected, then only byte compaction
155    * is used.
156    *
157    * @param msg the message
158    * @param compaction compaction mode to use
159    * @param encoding character encoding used to encode in default or byte compaction
160    *  or {@code null} for default / not applicable
161    * @return the encoded message (the char values range from 0 to 928)
162    */
163   static String encodeHighLevel(String msg, Compaction compaction, Charset encoding) throws WriterException {
164 
165     //the codewords 0..928 are encoded as Unicode characters
166     StringBuilder sb = new StringBuilder(msg.length());
167 
168     if (encoding == null) {
169       encoding = DEFAULT_ENCODING;
170     } else if (!DEFAULT_ENCODING.equals(encoding)) {
171       CharacterSetECI eci = CharacterSetECI.getCharacterSetECIByName(encoding.name());
172       if (eci != null) {
173         encodingECI(eci.getValue(), sb);
174       }
175     }
176 
177     int len = msg.length();
178     int p = 0;
179     int textSubMode = SUBMODE_ALPHA;
180 
181     // User selected encoding mode
182     if (compaction == Compaction.TEXT) {
183       encodeText(msg, p, len, sb, textSubMode);
184 
185     } else if (compaction == Compaction.BYTE) {
186       byte[] bytes = msg.getBytes(encoding);
187       encodeBinary(bytes, p, bytes.length, BYTE_COMPACTION, sb);
188 
189     } else if (compaction == Compaction.NUMERIC) {
190       sb.append((char) LATCH_TO_NUMERIC);
191       encodeNumeric(msg, p, len, sb);
192 
193     } else {
194       int encodingMode = TEXT_COMPACTION; //Default mode, see 4.4.2.1
195       while (p < len) {
196         int n = determineConsecutiveDigitCount(msg, p);
197         if (n >= 13) {
198           sb.append((char) LATCH_TO_NUMERIC);
199           encodingMode = NUMERIC_COMPACTION;
200           textSubMode = SUBMODE_ALPHA; //Reset after latch
201           encodeNumeric(msg, p, n, sb);
202           p += n;
203         } else {
204           int t = determineConsecutiveTextCount(msg, p);
205           if (t >= 5 || n == len) {
206             if (encodingMode != TEXT_COMPACTION) {
207               sb.append((char) LATCH_TO_TEXT);
208               encodingMode = TEXT_COMPACTION;
209               textSubMode = SUBMODE_ALPHA; //start with submode alpha after latch
210             }
211             textSubMode = encodeText(msg, p, t, sb, textSubMode);
212             p += t;
213           } else {
214             int b = determineConsecutiveBinaryCount(msg, p, encoding);
215             if (b == 0) {
216               b = 1;
217             }
218             byte[] bytes = msg.substring(p, p + b).getBytes(encoding);
219             if (bytes.length == 1 && encodingMode == TEXT_COMPACTION) {
220               //Switch for one byte (instead of latch)
221               encodeBinary(bytes, 0, 1, TEXT_COMPACTION, sb);
222             } else {
223               //Mode latch performed by encodeBinary()
224               encodeBinary(bytes, 0, bytes.length, encodingMode, sb);
225               encodingMode = BYTE_COMPACTION;
226               textSubMode = SUBMODE_ALPHA; //Reset after latch
227             }
228             p += b;
229           }
230         }
231       }
232     }
233 
234     return sb.toString();
235   }

237   /**
238    * Encode parts of the message using Text Compaction as described in ISO/IEC 15438:2001(E),
239    * chapter 4.4.2.
240    *
241    * @param msg            the message
242    * @param startpos       the start position within the message
243    * @param count          the number of characters to encode
244    * @param sb             receives the encoded codewords
245    * @param initialSubmode should normally be SUBMODE_ALPHA
246    * @return the text submode in which this method ends
247    */
248   private static int encodeText(CharSequence msg,
249                                 int startpos,
250                                 int count,
251                                 StringBuilder sb,
252                                 int initialSubmode) {
253     StringBuilder tmp = new StringBuilder(count);
254     int submode = initialSubmode;
255     int idx = 0;
256     while (true) {
257       char ch = msg.charAt(startpos + idx);
258       switch (submode) {
259         case SUBMODE_ALPHA:
260           if (isAlphaUpper(ch)) {
261             if (ch == ' ') {
262               tmp.append((char) 26); //space
263             } else {
264               tmp.append((char) (ch - 65));
265             }
266           } else {
267             if (isAlphaLower(ch)) {
268               submode = SUBMODE_LOWER;
269               tmp.append((char) 27); //ll
270               continue;
271             } else if (isMixed(ch)) {
272               submode = SUBMODE_MIXED;
273               tmp.append((char) 28); //ml
274               continue;
275             } else {
276               tmp.append((char) 29); //ps
277               tmp.append((char) PUNCTUATION[ch]);
278               break;
279             }
280           }
281           break;
282         case SUBMODE_LOWER:
283           if (isAlphaLower(ch)) {
284             if (ch == ' ') {
285               tmp.append((char) 26); //space
286             } else {
287               tmp.append((char) (ch - 97));
288             }
289           } else {
290             if (isAlphaUpper(ch)) {
291               tmp.append((char) 27); //as
292               tmp.append((char) (ch - 65));
293               //space cannot happen here, it is also in "Lower"
294               break;
295             } else if (isMixed(ch)) {
296               submode = SUBMODE_MIXED;
297               tmp.append((char) 28); //ml
298               continue;
299             } else {
300               tmp.append((char) 29); //ps
301               tmp.append((char) PUNCTUATION[ch]);
302               break;
303             }
304           }
305           break;
306         case SUBMODE_MIXED:
307           if (isMixed(ch)) {
308             tmp.append((char) MIXED[ch]);
309           } else {
310             if (isAlphaUpper(ch)) {
311               submode = SUBMODE_ALPHA;
312               tmp.append((char) 28); //al
313               continue;
314             } else if (isAlphaLower(ch)) {
315               submode = SUBMODE_LOWER;
316               tmp.append((char) 27); //ll
317               continue;
318             } else {
319               if (startpos + idx + 1 < count) {
320                 char next = msg.charAt(startpos + idx + 1);
321                 if (isPunctuation(next)) {
322                   submode = SUBMODE_PUNCTUATION;
323                   tmp.append((char) 25); //pl
324                   continue;
325                 }
326               }
327               tmp.append((char) 29); //ps
328               tmp.append((char) PUNCTUATION[ch]);
329             }
330           }
331           break;
332         default: //SUBMODE_PUNCTUATION
333           if (isPunctuation(ch)) {
334             tmp.append((char) PUNCTUATION[ch]);
335           } else {
336             submode = SUBMODE_ALPHA;
337             tmp.append((char) 29); //al
338             continue;
339           }
340       }
341       idx++;
342       if (idx >= count) {
343         break;
344       }
345     }
346     char h = 0;
347     int len = tmp.length();
348     for (int i = 0; i < len; i++) {
349       boolean odd = (i % 2) != 0;
350       if (odd) {
351         h = (char) ((h * 30) + tmp.charAt(i));
352         sb.append(h);
353       } else {
354         h = tmp.charAt(i);
355       }
356     }
357     if ((len % 2) != 0) {
358       sb.append((char) ((h * 30) + 29)); //ps
359     }
360     return submode;
361   }

363   /**
364    * Encode parts of the message using Byte Compaction as described in ISO/IEC 15438:2001(E),
365    * chapter 4.4.3. The Unicode characters will be converted to binary using the cp437
366    * codepage.
367    *
368    * @param bytes     the message converted to a byte array
369    * @param startpos  the start position within the message
370    * @param count     the number of bytes to encode
371    * @param startmode the mode from which this method starts
372    * @param sb        receives the encoded codewords
373    */
374   private static void encodeBinary(byte[] bytes,
375                                    int startpos,
376                                    int count,
377                                    int startmode,
378                                    StringBuilder sb) {
379     if (count == 1 && startmode == TEXT_COMPACTION) {
380       sb.append((char) SHIFT_TO_BYTE);
381     } else {
382       boolean sixpack = ((count % 6) == 0);
383       if (sixpack) {
384         sb.append((char)LATCH_TO_BYTE);
385       } else {
386         sb.append((char)LATCH_TO_BYTE_PADDED);
387       }
388     }
389 
390     int idx = startpos;
391     // Encode sixpacks
392     if (count >= 6) {
393       char[] chars = new char[5];
394       while ((startpos + count - idx) >= 6) {
395         long t = 0;
396         for (int i = 0; i < 6; i++) {
397           t <<= 8;
398           t += bytes[idx + i] & 0xff;
399         }
400         for (int i = 0; i < 5; i++) {
401           chars[i] = (char) (t % 900);
402           t /= 900;
403         }
404         for (int i = chars.length - 1; i >= 0; i--) {
405           sb.append(chars[i]);
406         }
407         idx += 6;
408       }
409     }
410     //Encode rest (remaining n<5 bytes if any)
411     for (int i = idx; i < startpos + count; i++) {
412       int ch = bytes[i] & 0xff;
413       sb.append((char) ch);
414     }
415   }

417   private static void encodeNumeric(String msg, int startpos, int count, StringBuilder sb) {
418     int idx = 0;
419     StringBuilder tmp = new StringBuilder(count / 3 + 1);
420     BigInteger num900 = BigInteger.valueOf(900);
421     BigInteger num0 = BigInteger.valueOf(0);
422     while (idx < count) {
423       tmp.setLength(0);
424       int len = Math.min(44, count - idx);
425       String part = '1' + msg.substring(startpos + idx, startpos + idx + len);
426       BigInteger bigint = new BigInteger(part);
427       do {
428         tmp.append((char) bigint.mod(num900).intValue());
429         bigint = bigint.divide(num900);
430       } while (!bigint.equals(num0));
431 
432       //Reverse temporary string
433       for (int i = tmp.length() - 1; i >= 0; i--) {
434         sb.append(tmp.charAt(i));
435       }
436       idx += len;
437     }
438   }

441   private static boolean isDigit(char ch) {
442     return ch >= '0' && ch <= '9';
443   }
444 
445   private static boolean isAlphaUpper(char ch) {
446     return ch == ' ' || (ch >= 'A' && ch <= 'Z');
447   }
448 
449   private static boolean isAlphaLower(char ch) {
450     return ch == ' ' || (ch >= 'a' && ch <= 'z');
451   }
452 
453   private static boolean isMixed(char ch) {
454     return MIXED[ch] != -1;
455   }
456 
457   private static boolean isPunctuation(char ch) {
458     return PUNCTUATION[ch] != -1;
459   }
460 
461   private static boolean isText(char ch) {
462     return ch == '\t' || ch == '\n' || ch == '\r' || (ch >= 32 && ch <= 126);
463   }

465   /**
466    * Determines the number of consecutive characters that are encodable using numeric compaction.
467    *
468    * @param msg      the message
469    * @param startpos the start position within the message
470    * @return the requested character count
471    */
472   private static int determineConsecutiveDigitCount(CharSequence msg, int startpos) {
473     int count = 0;
474     int len = msg.length();
475     int idx = startpos;
476     if (idx < len) {
477       char ch = msg.charAt(idx);
478       while (isDigit(ch) && idx < len) {
479         count++;
480         idx++;
481         if (idx < len) {
482           ch = msg.charAt(idx);
483         }
484       }
485     }
486     return count;
487   }

489   /**
490    * Determines the number of consecutive characters that are encodable using text compaction.
491    *
492    * @param msg      the message
493    * @param startpos the start position within the message
494    * @return the requested character count
495    */
496   private static int determineConsecutiveTextCount(CharSequence msg, int startpos) {
497     int len = msg.length();
498     int idx = startpos;
499     while (idx < len) {
500       char ch = msg.charAt(idx);
501       int numericCount = 0;
502       while (numericCount < 13 && isDigit(ch) && idx < len) {
503         numericCount++;
504         idx++;
505         if (idx < len) {
506           ch = msg.charAt(idx);
507         }
508       }
509       if (numericCount >= 13) {
510         return idx - startpos - numericCount;
511       }
512       if (numericCount > 0) {
513         //Heuristic: All text-encodable chars or digits are binary encodable
514         continue;
515       }
516       ch = msg.charAt(idx);
517 
518       //Check if character is encodable
519       if (!isText(ch)) {
520         break;
521       }
522       idx++;
523     }
524     return idx - startpos;
525   }

527   /**
528    * Determines the number of consecutive characters that are encodable using binary compaction.
529    *
530    * @param msg      the message
531    * @param startpos the start position within the message
532    * @param encoding the charset used to convert the message to a byte array
533    * @return the requested character count
534    */
535   private static int determineConsecutiveBinaryCount(String msg, int startpos, Charset encoding)
536       throws WriterException {
537     final CharsetEncoder encoder = encoding.newEncoder();
538     int len = msg.length();
539     int idx = startpos;
540     while (idx < len) {
541       char ch = msg.charAt(idx);
542       int numericCount = 0;
543 
544       while (numericCount < 13 && isDigit(ch)) {
545         numericCount++;
546         //textCount++;
547         int i = idx + numericCount;
548         if (i >= len) {
549           break;
550         }
551         ch = msg.charAt(i);
552       }
553       if (numericCount >= 13) {
554         return idx - startpos;
555       }
556       ch = msg.charAt(idx);
557 
558       if (!encoder.canEncode(ch)) {
559         throw new WriterException("Non-encodable character detected: " + ch + " (Unicode: " + (int) ch + ')');
560       }
561       idx++;
562     }
563     return idx - startpos;
564   }

566   private static void encodingECI(int eci, StringBuilder sb) throws WriterException {
567     if (eci >= 0 && eci < 900) {
568       sb.append((char) ECI_CHARSET);
569       sb.append((char) eci);
570     } else if (eci < 810900) {
571       sb.append((char) ECI_GENERAL_PURPOSE);
572       sb.append((char) (eci / 900 - 1));
573       sb.append((char) (eci % 900));
574     } else if (eci < 811800) {
575       sb.append((char) ECI_USER_DEFINED);
576       sb.append((char) (810900 - eci));
577     } else {
578       throw new WriterException("ECI number not in valid range from 0..811799, but was " + eci);
579     }
580   }
