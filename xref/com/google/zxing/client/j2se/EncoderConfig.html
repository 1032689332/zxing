17  package com.google.zxing.client.j2se;
18  
19  import java.util.List;
20  
21  import com.beust.jcommander.Parameter;
22  import com.beust.jcommander.validators.PositiveInteger;
23  import com.google.zxing.BarcodeFormat;
24  
25  final class EncoderConfig {
26  
27    static final String DEFAULT_OUTPUT_FILE_BASE = "out";
28  
29    @Parameter(names = "--barcode_format",
30        description = "Format to encode, from BarcodeFormat class. Not all formats are supported")
31    BarcodeFormat barcodeFormat = BarcodeFormat.QR_CODE;
32  
33    @Parameter(names = "--image_format",
34        description = "image output format, such as PNG, JPG, GIF")
35    String imageFormat = "PNG";
36  
37    @Parameter(names = "--output",
38        description = " File to write to. Defaults to out.png")
39    String outputFileBase = DEFAULT_OUTPUT_FILE_BASE;
40  
41    @Parameter(names = "--width",
42        description = "Image width",
43        validateWith = PositiveInteger.class)
44    int width = 300;
45  
46    @Parameter(names = "--height",
47        description = "Image height",
48        validateWith = PositiveInteger.class)
49    int height = 300;
50  
51    @Parameter(names = "--help",
52        description = "Prints this help message",
53        help = true)
54    boolean help;
55  
56    @Parameter(description = "(Text to encode)", required = true)
57    List<String> contents;
58  
59  }
