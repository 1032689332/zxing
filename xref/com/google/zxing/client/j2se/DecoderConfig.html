17  package com.google.zxing.client.j2se;
18  
19  import com.beust.jcommander.Parameter;
20  import com.beust.jcommander.validators.PositiveInteger;
21  import com.google.zxing.BarcodeFormat;
22  import com.google.zxing.DecodeHintType;
23  
24  import java.net.URI;
25  import java.util.ArrayList;
26  import java.util.Arrays;
27  import java.util.Collections;
28  import java.util.EnumMap;
29  import java.util.List;
30  import java.util.Map;
31  
32  final class DecoderConfig {
33  
34    @Parameter(names = "--try_harder",
35        description = "Use the TRY_HARDER hint, default is normal mode")
36    boolean tryHarder;
37  
38    @Parameter(names="--pure_barcode",
39        description="Input image is a pure monochrome barcode image, not a photo")
40    boolean pureBarcode;
41  
42    @Parameter(names = "--products_only",
43        description = "Only decode the UPC and EAN families of barcodes")
44    boolean productsOnly;
45  
46    @Parameter(names = "--dump_results",
47        description = "Write the decoded contents to input.txt")
48    boolean dumpResults;
49  
50    @Parameter(names = "--dump_black_point",
51        description = "Compare black point algorithms with dump as input.mono.png")
52    boolean dumpBlackPoint;
53  
54    @Parameter(names = "--multi",
55        description = "Scans image for multiple barcodes")
56    boolean multi;
57  
58    @Parameter(names = "--brief",
59        description = "Only output one line per file, omitting the contents")
60    boolean brief;
61  
62    @Parameter(names = "--recursive",
63        description = "Descend into subdirectories")
64    boolean recursive;
65  
66    @Parameter(names = "--crop",
67        description = " Only examine cropped region of input image(s)",
68        arity = 4,
69        validateWith = PositiveInteger.class)
70    List<Integer> crop;
71  
72    @Parameter(names = "--possible_formats",
73        description = "Formats to decode, where format is any value in BarcodeFormat",
74        variableArity = true)
75    List<BarcodeFormat> possibleFormats;
76  
77    @Parameter(names = "--help",
78        description = "Prints this help message",
79        help = true)
80    boolean help;
81  
82    @Parameter(description = "(URIs to decode)", required = true, variableArity = true)
83    List<URI> inputPaths;
84  
85    Map<DecodeHintType,?> buildHints() {
86      List<BarcodeFormat> finalPossibleFormats = possibleFormats;
87      if (finalPossibleFormats == null || finalPossibleFormats.isEmpty()) {
88        finalPossibleFormats = new ArrayList<>();
89        finalPossibleFormats.addAll(Arrays.asList(
90            BarcodeFormat.UPC_A,
91            BarcodeFormat.UPC_E,
92            BarcodeFormat.EAN_13,
93            BarcodeFormat.EAN_8,
94            BarcodeFormat.RSS_14,
95            BarcodeFormat.RSS_EXPANDED
96        ));
97        if (!productsOnly) {
98          finalPossibleFormats.addAll(Arrays.asList(
99              BarcodeFormat.CODE_39,
100             BarcodeFormat.CODE_93,
101             BarcodeFormat.CODE_128,
102             BarcodeFormat.ITF,
103             BarcodeFormat.QR_CODE,
104             BarcodeFormat.DATA_MATRIX,
105             BarcodeFormat.AZTEC,
106             BarcodeFormat.PDF_417,
107             BarcodeFormat.CODABAR,
108             BarcodeFormat.MAXICODE
109         ));
110       }
111     }
112 
113     Map<DecodeHintType, Object> hints = new EnumMap<>(DecodeHintType.class);
114     hints.put(DecodeHintType.POSSIBLE_FORMATS, finalPossibleFormats);
115     if (tryHarder) {
116       hints.put(DecodeHintType.TRY_HARDER, Boolean.TRUE);
117     }
118     if (pureBarcode) {
119       hints.put(DecodeHintType.PURE_BARCODE, Boolean.TRUE);
120     }
121     return Collections.unmodifiableMap(hints);
122   }
123 
124 }
