29  /**
30   * This produces nearly optimal encodings of text into the first-level of
31   * encoding used by Aztec code.
32   *
33   * It uses a dynamic algorithm.  For each prefix of the string, it determines
34   * a set of encodings that could lead to this prefix.  We repeatedly add a
35   * character and generate a new set of optimal encodings until we have read
36   * through the entire input.
37   *
38   * @author Frank Yellin
39   * @author Rustam Abdullaev
40   */
41  public final class HighLevelEncoder {
42  
43    static final String[] MODE_NAMES = {"UPPER", "LOWER", "DIGIT", "MIXED", "PUNCT"};
44  
45    static final int MODE_UPPER = 0; // 5 bits
46    static final int MODE_LOWER = 1; // 5 bits
47    static final int MODE_DIGIT = 2; // 4 bits
48    static final int MODE_MIXED = 3; // 5 bits
49    static final int MODE_PUNCT = 4; // 5 bits
50  
51    // The Latch Table shows, for each pair of Modes, the optimal method for
52    // getting from one mode to another.  In the worst possible case, this can
53    // be up to 14 bits.  In the best possible case, we are already there!
54    // The high half-word of each entry gives the number of bits.
55    // The low half-word of each entry are the actual bits necessary to change
56    static final int[][] LATCH_TABLE = {
57      {
58        0,
59        (5 << 16) + 28,              // UPPER -> LOWER
60        (5 << 16) + 30,              // UPPER -> DIGIT
61        (5 << 16) + 29,              // UPPER -> MIXED
62        (10 << 16) + (29 << 5) + 30, // UPPER -> MIXED -> PUNCT
63      },
64      {
65        (9 << 16) + (30 << 4) + 14,  // LOWER -> DIGIT -> UPPER
66        0,
67        (5 << 16) + 30,              // LOWER -> DIGIT
68        (5 << 16) + 29,              // LOWER -> MIXED
69        (10 << 16) + (29 << 5) + 30, // LOWER -> MIXED -> PUNCT
70      },
71      {
72        (4 << 16) + 14,              // DIGIT -> UPPER
73        (9 << 16) + (14 << 5) + 28,  // DIGIT -> UPPER -> LOWER
74        0,
75        (9 << 16) + (14 << 5) + 29,  // DIGIT -> UPPER -> MIXED
76        (14 << 16) + (14 << 10) + (29 << 5) + 30,
77                                     // DIGIT -> UPPER -> MIXED -> PUNCT
78      },
79      {
80        (5 << 16) + 29,              // MIXED -> UPPER
81        (5 << 16) + 28,              // MIXED -> LOWER
82        (10 << 16) + (29 << 5) + 30, // MIXED -> UPPER -> DIGIT
83        0,
84        (5 << 16) + 30,              // MIXED -> PUNCT
85      },
86      {
87        (5 << 16) + 31,              // PUNCT -> UPPER
88        (10 << 16) + (31 << 5) + 28, // PUNCT -> UPPER -> LOWER
89        (10 << 16) + (31 << 5) + 30, // PUNCT -> UPPER -> DIGIT
90        (10 << 16) + (31 << 5) + 29, // PUNCT -> UPPER -> MIXED
91        0,
92      },
93    };
94  
95    // A reverse mapping from [mode][char] to the encoding for that character
96    // in that mode.  An entry of 0 indicates no mapping exists.
97    private static final int[][] CHAR_MAP = new int[5][256];
98    static {
99      CHAR_MAP[MODE_UPPER][' '] = 1;
100     for (int c = 'A'; c <= 'Z'; c++) {
101       CHAR_MAP[MODE_UPPER][c] = c - 'A' + 2;
102     }
103     CHAR_MAP[MODE_LOWER][' '] = 1;
104     for (int c = 'a'; c <= 'z'; c++) {
105       CHAR_MAP[MODE_LOWER][c] = c - 'a' + 2;
106     }
107     CHAR_MAP[MODE_DIGIT][' '] = 1;
108     for (int c = '0'; c <= '9'; c++) {
109       CHAR_MAP[MODE_DIGIT][c] = c - '0' + 2;
110     }
111     CHAR_MAP[MODE_DIGIT][','] = 12;
112     CHAR_MAP[MODE_DIGIT]['.'] = 13;
113     int[] mixedTable = {
114         '\0', ' ', '\1', '\2', '\3', '\4', '\5', '\6', '\7', '\b', '\t', '\n',
115         '\13', '\f', '\r', '\33', '\34', '\35', '\36', '\37', '@', '\\', '^',
116         '_', '`', '|', '~', '\177'
117     };
118     for (int i = 0; i < mixedTable.length; i++) {
119       CHAR_MAP[MODE_MIXED][mixedTable[i]] = i;
120     }
121     int[] punctTable = {
122         '\0', '\r', '\0', '\0', '\0', '\0', '!', '\'', '#', '$', '%', '&', '\'',
123         '(', ')', '*', '+', ',', '-', '.', '/', ':', ';', '<', '=', '>', '?',
124         '[', ']', '{', '}'
125     };
126     for (int i = 0; i < punctTable.length; i++) {
127       if (punctTable[i] > 0) {
128         CHAR_MAP[MODE_PUNCT][punctTable[i]] = i;
129       }
130     }
131   }

133   // A map showing the available shift codes.  (The shifts to BINARY are not
134   // shown
135   static final int[][] SHIFT_TABLE = new int[6][6]; // mode shift codes, per table
136   static {
137     for (int[] table : SHIFT_TABLE) {
138       Arrays.fill(table, -1);
139     }
140     SHIFT_TABLE[MODE_UPPER][MODE_PUNCT] = 0;
141 
142     SHIFT_TABLE[MODE_LOWER][MODE_PUNCT] = 0;
143     SHIFT_TABLE[MODE_LOWER][MODE_UPPER] = 28;
144 
145     SHIFT_TABLE[MODE_MIXED][MODE_PUNCT] = 0;
146 
147     SHIFT_TABLE[MODE_DIGIT][MODE_PUNCT] = 0;
148     SHIFT_TABLE[MODE_DIGIT][MODE_UPPER] = 15;
149   }

153   public HighLevelEncoder(byte[] text) {

157   /**
158    * @return text represented by this encoder encoded as a {@link BitArray}
159    */
160   public BitArray encode() {
161     Collection<State> states = Collections.singletonList(State.INITIAL_STATE);
162     for (int index = 0; index < text.length; index++) {
163       int pairCode;
164       int nextChar = index + 1 < text.length ? text[index + 1] : 0;
165       switch (text[index]) {
166         case '\r':  
167           pairCode = nextChar == '\n' ? 2 : 0; 
168           break;
169         case '.' :  
170           pairCode = nextChar == ' '  ? 3 : 0; 
171           break;
172         case ',' :  
173           pairCode = nextChar == ' ' ? 4 : 0; 
174           break;
175         case ':' :  
176           pairCode = nextChar == ' ' ? 5 : 0; 
177           break;
178         default:    
179           pairCode = 0;
180       }
181       if (pairCode > 0) {
182         // We have one of the four special PUNCT pairs.  Treat them specially.
183         // Get a new set of states for the two new characters.
184         states = updateStateListForPair(states, index, pairCode);
185         index++;
186       } else {
187         // Get a new set of states for the new character.
188         states = updateStateListForChar(states, index);
189       }
190     }
191     // We are left with a set of states.  Find the shortest one.
192     State minState = Collections.min(states, new Comparator<State>() {
193       @Override
194       public int compare(State a, State b) {
195         return a.getBitCount() - b.getBitCount();
196       }
197     });
198     // Convert it to a bit array, and return.
199     return minState.toBitArray(text);
200   }
201 
202   // We update a set of states for a new character by updating each state
203   // for the new character, merging the results, and then removing the
204   // non-optimal states.
205   private Collection<State> updateStateListForChar(Iterable<State> states, int index) {
206     Collection<State> result = new LinkedList<>();
207     for (State state : states) {
208       updateStateForChar(state, index, result);
209     }
210     return simplifyStates(result);
211   }
212 
213   // Return a set of states that represent the possible ways of updating this
214   // state for the next character.  The resulting set of states are added to
215   // the "result" list.
216   private void updateStateForChar(State state, int index, Collection<State> result) {
217     char ch = (char) (text[index] & 0xFF);
218     boolean charInCurrentTable = CHAR_MAP[state.getMode()][ch] > 0;
219     State stateNoBinary = null;
220     for (int mode = 0; mode <= MODE_PUNCT; mode++) {
221       int charInMode = CHAR_MAP[mode][ch];
222       if (charInMode > 0) {
223         if (stateNoBinary == null) {
224           // Only create stateNoBinary the first time it's required.
225           stateNoBinary = state.endBinaryShift(index);
226         }
227         // Try generating the character by latching to its mode
228         if (!charInCurrentTable || mode == state.getMode() || mode == MODE_DIGIT) {
229           // If the character is in the current table, we don't want to latch to
230           // any other mode except possibly digit (which uses only 4 bits).  Any
231           // other latch would be equally successful *after* this character, and
232           // so wouldn't save any bits.
233           State latchState = stateNoBinary.latchAndAppend(mode, charInMode);
234           result.add(latchState);
235         }
236         // Try generating the character by switching to its mode.
237         if (!charInCurrentTable && SHIFT_TABLE[state.getMode()][mode] >= 0) {
238           // It never makes sense to temporarily shift to another mode if the
239           // character exists in the current mode.  That can never save bits.
240           State shiftState = stateNoBinary.shiftAndAppend(mode, charInMode);
241           result.add(shiftState);
242         }
243       }
244     }
245     if (state.getBinaryShiftByteCount() > 0 || CHAR_MAP[state.getMode()][ch] == 0) {
246       // It's never worthwhile to go into binary shift mode if you're not already
247       // in binary shift mode, and the character exists in your current mode.
248       // That can never save bits over just outputting the char in the current mode.
249       State binaryState = state.addBinaryShiftChar(index);
250       result.add(binaryState);
251     }
252   }

254   private static Collection<State> updateStateListForPair(Iterable<State> states, int index, int pairCode) {
255     Collection<State> result = new LinkedList<>();
256     for (State state : states) {
257       updateStateForPair(state, index, pairCode, result);
258     }
259     return simplifyStates(result);
260   }
261 
262   private static void updateStateForPair(State state, int index, int pairCode, Collection<State> result) {
263     State stateNoBinary = state.endBinaryShift(index);
264     // Possibility 1.  Latch to MODE_PUNCT, and then append this code
265     result.add(stateNoBinary.latchAndAppend(MODE_PUNCT, pairCode));
266     if (state.getMode() != MODE_PUNCT) {
267       // Possibility 2.  Shift to MODE_PUNCT, and then append this code.
268       // Every state except MODE_PUNCT (handled above) can shift
269       result.add(stateNoBinary.shiftAndAppend(MODE_PUNCT, pairCode));
270     }
271     if (pairCode == 3 || pairCode == 4) {
272       // both characters are in DIGITS.  Sometimes better to just add two digits
273       State digitState = stateNoBinary
274           .latchAndAppend(MODE_DIGIT, 16 - pairCode)  // period or comma in DIGIT
275           .latchAndAppend(MODE_DIGIT, 1);             // space in DIGIT
276       result.add(digitState);
277     }
278     if (state.getBinaryShiftByteCount() > 0) {
279       // It only makes sense to do the characters as binary if we're already
280       // in binary mode.
281       State binaryState = state.addBinaryShiftChar(index).addBinaryShiftChar(index + 1);
282       result.add(binaryState);
283     }
284   }
285 
286   private static Collection<State> simplifyStates(Iterable<State> states) {
287     List<State> result = new LinkedList<>();
288     for (State newState : states) {
289       boolean add = true;
290       for (Iterator<State> iterator = result.iterator(); iterator.hasNext();) {
291         State oldState = iterator.next();
292         if (oldState.isBetterThanOrEqualTo(newState)) {
293           add = false;
294           break;
295         }
296         if (newState.isBetterThanOrEqualTo(oldState)) {
297           iterator.remove();
298         }
299       }
300       if (add) {
301         result.add(newState);
302       }
303     }
304     return result;
305   }
