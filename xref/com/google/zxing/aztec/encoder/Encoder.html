31    public static final int DEFAULT_EC_PERCENT = 33; // default minimal percentage of error check words
32    public static final int DEFAULT_AZTEC_LAYERS = 0;
33    private static final int MAX_NB_BITS = 32;
34    private static final int MAX_NB_BITS_COMPACT = 4;
35  
36    private static final int[] WORD_SIZE = {
37      4, 6, 6, 8, 8, 8, 8, 8, 8, 10, 10, 10, 10, 10, 10, 10, 10, 10, 10, 10, 10, 10, 10,
38      12, 12, 12, 12, 12, 12, 12, 12, 12, 12
39    };

42    }

44    /**
45     * Encodes the given binary content as an Aztec symbol
46     * 
47     * @param data input data string
48     * @return Aztec symbol matrix with metadata
49     */
50    public static AztecCode encode(byte[] data) {
51      return encode(data, DEFAULT_EC_PERCENT, DEFAULT_AZTEC_LAYERS);
52    }
53    
54    /**
55     * Encodes the given binary content as an Aztec symbol
56     * 
57     * @param data input data string
58     * @param minECCPercent minimal percentage of error check words (According to ISO/IEC 24778:2008,
59     *                      a minimum of 23% + 3 words is recommended)
60     * @param userSpecifiedLayers if non-zero, a user-specified value for the number of layers
61     * @return Aztec symbol matrix with metadata
62     */
63    public static AztecCode encode(byte[] data, int minECCPercent, int userSpecifiedLayers) {
64      // High-level encode
65      BitArray bits = new HighLevelEncoder(data).encode();
66      
67      // stuff bits and choose symbol size
68      int eccBits = bits.getSize() * minECCPercent / 100 + 11;
69      int totalSizeBits = bits.getSize() + eccBits;
70      boolean compact;
71      int layers;
72      int totalBitsInLayer;
73      int wordSize;
74      BitArray stuffedBits;
75      if (userSpecifiedLayers != DEFAULT_AZTEC_LAYERS) {
76        compact = userSpecifiedLayers < 0;
77        layers = Math.abs(userSpecifiedLayers);
78        if (layers > (compact ? MAX_NB_BITS_COMPACT : MAX_NB_BITS)) {
79          throw new IllegalArgumentException(
80              String.format("Illegal value %s for layers", userSpecifiedLayers));
81        }
82        totalBitsInLayer = totalBitsInLayer(layers, compact);
83        wordSize = WORD_SIZE[layers];
84        int usableBitsInLayers = totalBitsInLayer - (totalBitsInLayer % wordSize);
85        stuffedBits = stuffBits(bits, wordSize);
86        if (stuffedBits.getSize() + eccBits > usableBitsInLayers) {
87          throw new IllegalArgumentException("Data to large for user specified layer");
88        }
89        if (compact && stuffedBits.getSize() > wordSize * 64) {
90          // Compact format only allows 64 data words, though C4 can hold more words than that
91          throw new IllegalArgumentException("Data to large for user specified layer");
92        }
93      } else {
94        wordSize = 0;
95        stuffedBits = null;
96        // We look at the possible table sizes in the order Compact1, Compact2, Compact3,
97        // Compact4, Normal4,...  Normal(i) for i < 4 isn't typically used since Compact(i+1)
98        // is the same size, but has more data.
99        for (int i = 0; ; i++) {
100         if (i > MAX_NB_BITS) {
101           throw new IllegalArgumentException("Data too large for an Aztec code");
102         }
103         compact = i <= 3;
104         layers = compact ? i + 1 : i;
105         totalBitsInLayer = totalBitsInLayer(layers, compact);
106         if (totalSizeBits > totalBitsInLayer) {
107           continue;
108         }
109         // [Re]stuff the bits if this is the first opportunity, or if the
110         // wordSize has changed
111         if (wordSize != WORD_SIZE[layers]) {
112           wordSize = WORD_SIZE[layers];
113           stuffedBits = stuffBits(bits, wordSize);
114         }
115         int usableBitsInLayers = totalBitsInLayer - (totalBitsInLayer % wordSize);
116         if (compact && stuffedBits.getSize() > wordSize * 64) {
117           // Compact format only allows 64 data words, though C4 can hold more words than that
118           continue;
119         }
120         if (stuffedBits.getSize() + eccBits <= usableBitsInLayers) {
121           break;
122         }
123       }
124     }
125     BitArray messageBits = generateCheckWords(stuffedBits, totalBitsInLayer, wordSize);
126     
127     // generate mode message
128     int messageSizeInWords = stuffedBits.getSize() / wordSize;
129     BitArray modeMessage = generateModeMessage(compact, layers, messageSizeInWords);
130 
131     // allocate symbol
132     int baseMatrixSize = compact ? 11 + layers * 4 : 14 + layers * 4; // not including alignment lines
133     int[] alignmentMap = new int[baseMatrixSize];
134     int matrixSize;
135     if (compact) {
136       // no alignment marks in compact mode, alignmentMap is a no-op
137       matrixSize = baseMatrixSize;
138       for (int i = 0; i < alignmentMap.length; i++) {
139         alignmentMap[i] = i;
140       }
141     } else {
142       matrixSize = baseMatrixSize + 1 + 2 * ((baseMatrixSize / 2 - 1) / 15);
143       int origCenter = baseMatrixSize / 2;
144       int center = matrixSize / 2;
145       for (int i = 0; i < origCenter; i++) {
146         int newOffset = i + i / 15;
147         alignmentMap[origCenter - i - 1] = center - newOffset - 1;
148         alignmentMap[origCenter + i] = center + newOffset + 1;
149       }
150     }
151     BitMatrix matrix = new BitMatrix(matrixSize);
152     
153     // draw data bits
154     for (int i = 0, rowOffset = 0; i < layers; i++) {
155       int rowSize = compact ? (layers - i) * 4 + 9 : (layers - i) * 4 + 12;
156       for (int j = 0; j < rowSize; j++) {
157         int columnOffset = j * 2;
158         for (int k = 0; k < 2; k++) {
159           if (messageBits.get(rowOffset + columnOffset + k)) {
160             matrix.set(alignmentMap[i * 2 + k], alignmentMap[i * 2 + j]);
161           }
162           if (messageBits.get(rowOffset + rowSize * 2 + columnOffset + k)) {
163             matrix.set(alignmentMap[i * 2 + j], alignmentMap[baseMatrixSize - 1 - i * 2 - k]);
164           }
165           if (messageBits.get(rowOffset + rowSize * 4 + columnOffset + k)) {
166             matrix.set(alignmentMap[baseMatrixSize - 1 - i * 2 - k], alignmentMap[baseMatrixSize - 1 - i * 2 - j]);
167           }
168           if (messageBits.get(rowOffset + rowSize * 6 + columnOffset + k)) {
169             matrix.set(alignmentMap[baseMatrixSize - 1 - i * 2 - j], alignmentMap[i * 2 + k]);
170           }
171         }
172       }
173       rowOffset += rowSize * 8;
174     }
175 
176     // draw mode message
177     drawModeMessage(matrix, compact, matrixSize, modeMessage);
178     
179     // draw alignment marks
180     if (compact) {
181       drawBullsEye(matrix, matrixSize / 2, 5);
182     } else {
183       drawBullsEye(matrix, matrixSize / 2, 7);
184       for (int i = 0, j = 0; i < baseMatrixSize / 2 - 1; i += 15, j += 16) {
185         for (int k = (matrixSize / 2) & 1; k < matrixSize; k += 2) {
186           matrix.set(matrixSize / 2 - j, k);
187           matrix.set(matrixSize / 2 + j, k);
188           matrix.set(k, matrixSize / 2 - j);
189           matrix.set(k, matrixSize / 2 + j);
190         }
191       }
192     }
193     
194     AztecCode aztec = new AztecCode();
195     aztec.setCompact(compact);
196     aztec.setSize(matrixSize);
197     aztec.setLayers(layers);
198     aztec.setCodeWords(messageSizeInWords);
199     aztec.setMatrix(matrix);
200     return aztec;
201   }

203   private static void drawBullsEye(BitMatrix matrix, int center, int size) {
204     for (int i = 0; i < size; i += 2) {
205       for (int j = center - i; j <= center + i; j++) {
206         matrix.set(j, center - i);
207         matrix.set(j, center + i);
208         matrix.set(center - i, j);
209         matrix.set(center + i, j);
210       }
211     }
212     matrix.set(center - size, center - size);
213     matrix.set(center - size + 1, center - size);
214     matrix.set(center - size, center - size + 1);
215     matrix.set(center + size, center - size);
216     matrix.set(center + size, center - size + 1);
217     matrix.set(center + size, center + size - 1);
218   }

220   static BitArray generateModeMessage(boolean compact, int layers, int messageSizeInWords) {
221     BitArray modeMessage = new BitArray();
222     if (compact) {
223       modeMessage.appendBits(layers - 1, 2);
224       modeMessage.appendBits(messageSizeInWords - 1, 6);
225       modeMessage = generateCheckWords(modeMessage, 28, 4);
226     } else {
227       modeMessage.appendBits(layers - 1, 5);
228       modeMessage.appendBits(messageSizeInWords - 1, 11);
229       modeMessage = generateCheckWords(modeMessage, 40, 4);
230     }
231     return modeMessage;
232   }
233   
234   private static void drawModeMessage(BitMatrix matrix, boolean compact, int matrixSize, BitArray modeMessage) {
235     int center = matrixSize / 2;
236     if (compact) {
237       for (int i = 0; i < 7; i++) {
238         int offset = center - 3 + i;
239         if (modeMessage.get(i)) {
240           matrix.set(offset, center - 5);
241         }
242         if (modeMessage.get(i + 7)) {
243           matrix.set(center + 5, offset);
244         }
245         if (modeMessage.get(20 - i)) {
246           matrix.set(offset, center + 5);
247         }
248         if (modeMessage.get(27 - i)) {
249           matrix.set(center - 5, offset);
250         }
251       }
252     } else {
253       for (int i = 0; i < 10; i++) {
254         int offset = center - 5 + i + i / 5;
255         if (modeMessage.get(i)) {
256           matrix.set(offset, center - 7);
257         }
258         if (modeMessage.get(i + 10)) {
259           matrix.set(center + 7, offset);
260         }
261         if (modeMessage.get(29 - i)) {
262           matrix.set(offset, center + 7);
263         }
264         if (modeMessage.get(39 - i)) {
265           matrix.set(center - 7, offset);
266         }
267       }
268     }
269   }

271   private static BitArray generateCheckWords(BitArray bitArray, int totalBits, int wordSize) {
272     // bitArray is guaranteed to be a multiple of the wordSize, so no padding needed
273     int messageSizeInWords = bitArray.getSize() / wordSize;
274     ReedSolomonEncoder rs = new ReedSolomonEncoder(getGF(wordSize));
275     int totalWords = totalBits / wordSize;
276     int[] messageWords = bitsToWords(bitArray, wordSize, totalWords);
277     rs.encode(messageWords, totalWords - messageSizeInWords);
278     int startPad = totalBits % wordSize;
279     BitArray messageBits = new BitArray();
280     messageBits.appendBits(0, startPad);
281     for (int messageWord : messageWords) {
282       messageBits.appendBits(messageWord, wordSize);
283     }
284     return messageBits;
285   }

287   private static int[] bitsToWords(BitArray stuffedBits, int wordSize, int totalWords) {
288     int[] message = new int[totalWords];
289     int i;
290     int n;
291     for (i = 0, n = stuffedBits.getSize() / wordSize; i < n; i++) {
292       int value = 0;
293       for (int j = 0; j < wordSize; j++) {
294         value |= stuffedBits.get(i * wordSize + j) ? (1 << wordSize - j - 1) : 0;
295       }
296       message[i] = value;
297     }
298     return message;
299   }

301   private static GenericGF getGF(int wordSize) {
302     switch (wordSize) {
303       case 4:
304         return GenericGF.AZTEC_PARAM;
305       case 6:
306         return GenericGF.AZTEC_DATA_6;
307       case 8:
308         return GenericGF.AZTEC_DATA_8;
309       case 10:
310         return GenericGF.AZTEC_DATA_10;
311       case 12:
312         return GenericGF.AZTEC_DATA_12;
313       default:
314         throw new IllegalArgumentException("Unsupported word size " + wordSize);
315     }
316   }

318   static BitArray stuffBits(BitArray bits, int wordSize) {
319     BitArray out = new BitArray();
320 
321     int n = bits.getSize();
322     int mask = (1 << wordSize) - 2;
323     for (int i = 0; i < n; i += wordSize) {
324       int word = 0;
325       for (int j = 0; j < wordSize; j++) {
326         if (i + j >= n || bits.get(i + j)) {
327           word |= 1 << (wordSize - 1 - j);
328         }
329       }
330       if ((word & mask) == mask) {
331         out.appendBits(word & mask, wordSize);
332         i--;
333       } else if ((word & mask) == 0) {
334         out.appendBits(word | 1, wordSize);
335         i--;
336       } else {
337         out.appendBits(word, wordSize);
338       }
339     }
340     return out;
341   }
342 
343   private static int totalBitsInLayer(int layers, boolean compact) {
344     return ((compact ? 88 : 112) + 16 * layers) * layers;
345   }
