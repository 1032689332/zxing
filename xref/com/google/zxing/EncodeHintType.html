17  package com.google.zxing;
18  
19  /**
20   * These are a set of hints that you may pass to Writers to specify their behavior.
21   *
22   * @author dswitkin@google.com (Daniel Switkin)
23   */
24  public enum EncodeHintType {
25  
26    /**
27     * Specifies what degree of error correction to use, for example in QR Codes.
28     * Type depends on the encoder. For example for QR codes it's type
29     * {@link com.google.zxing.qrcode.decoder.ErrorCorrectionLevel ErrorCorrectionLevel}.
30     * For Aztec it is of type {@link Integer}, representing the minimal percentage of error correction words.
31     * For PDF417 it is of type {@link Integer}, valid values being 0 to 8.
32     * Note: an Aztec symbol should have a minimum of 25% EC words.
33     */
34    ERROR_CORRECTION,
35  
36    /**
37     * Specifies what character encoding to use where applicable (type {@link String})
38     */
39    CHARACTER_SET,
40  
41    /**
42     * Specifies the matrix shape for Data Matrix (type {@link com.google.zxing.datamatrix.encoder.SymbolShapeHint})
43     */
44    DATA_MATRIX_SHAPE,
45  
46    /**
47     * Specifies a minimum barcode size (type {@link Dimension}). Only applicable to Data Matrix now.
48     *
49     * @deprecated use width/height params in
50     * {@link com.google.zxing.datamatrix.DataMatrixWriter#encode(String, BarcodeFormat, int, int)}
51     */
52    @Deprecated
53    MIN_SIZE,
54  
55    /**
56     * Specifies a maximum barcode size (type {@link Dimension}). Only applicable to Data Matrix now.
57     *
58     * @deprecated without replacement
59     */
60    @Deprecated
61    MAX_SIZE,
62  
63    /**
64     * Specifies margin, in pixels, to use when generating the barcode. The meaning can vary
65     * by format; for example it controls margin before and after the barcode horizontally for
66     * most 1D formats. (Type {@link Integer}).
67     */
68    MARGIN,
69  
70    /**
71     * Specifies whether to use compact mode for PDF417 (type {@link Boolean}).
72     */
73    PDF417_COMPACT,
74  
75    /**
76     * Specifies what compaction mode to use for PDF417 (type
77     * {@link com.google.zxing.pdf417.encoder.Compaction Compaction}).
78     */
79    PDF417_COMPACTION,
80  
81    /**
82     * Specifies the minimum and maximum number of rows and columns for PDF417 (type
83     * {@link com.google.zxing.pdf417.encoder.Dimensions Dimensions}).
84     */
85    PDF417_DIMENSIONS,
86  
87    /**
88     * Specifies the required number of layers for an Aztec code:
89     *   a negative number (-1, -2, -3, -4) specifies a compact Aztec code
90     *   0 indicates to use the minimum number of layers (the default)
91     *   a positive number (1, 2, .. 32) specifies a normaol (non-compact) Aztec code
92     */
93     AZTEC_LAYERS,
94  }
