46    private static final String[] FULL_DAY_ONLY_IDS =
47        { "fullDayOnlyInfo1", "fullDayOnlyInfo2", "fullDayOnlyInfo3", "fullDayOnlyInfo4" };
48    private static final long ONE_HOUR = 60L * 60 * 1000;
49  
50    private Grid table;
51    private final TextBox eventName = new TextBox();
52    private final CheckBox fullDay = new CheckBox();
53    private final DatePicker datePicker1 = new DatePicker();
54    private final DatePicker datePicker2 = new DatePicker();
55    private final TextBox timePicker1 = new TextBox();
56    private final TextBox timePicker2 = new TextBox();
57    private final CheckBox summerTime = new CheckBox();
58    private final ListBox timeZones = new ListBox();
59    private Date timePicker1PreviousDate = null;
60    private final TextBox location = new TextBox();
61    private final TextBox description = new TextBox();

63    public CalendarEventGenerator(final ChangeHandler handler, KeyPressHandler keyListener) {
64      eventName.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
65      eventName.addChangeHandler(handler);
66      eventName.addKeyPressHandler(keyListener);
67      setDateToTextBox(timePicker1, new Date());
68      try {
69        setDateToTextBox(timePicker2, addMilliseconds(getDateFromTextBox(timePicker1), ONE_HOUR));
70        timePicker1PreviousDate = getDateFromTextBox(timePicker1);
71      } catch (GeneratorException ge) {
72        throw new IllegalStateException(ge);
73      }
74  
75      buildTimeZoneList();
76      timeZones.setSelectedIndex(25);
77      timeZones.addKeyPressHandler(keyListener);
78      timePicker1.addValueChangeHandler(new ValueChangeHandler<String>() {
79        @Override
80        public void onValueChange(ValueChangeEvent<String> valueChangeEvent) {
81          Date time = timePicker1PreviousDate;
82          Date time1;
83          Date time2;
84          try {
85            time1 = getDateFromTextBox(timePicker1);
86            time2 = getDateFromTextBox(timePicker2);
87          } catch (GeneratorException e) {
88            return;
89          }
90          if (time2.after(time)) {
91            // keep the same time difference if the interval is valid.
92            long diff = time2.getTime() - time.getTime();
93            setDateToTextBox(timePicker2, addMilliseconds(time1, diff));
94          } else {
95            // otherwise erase the end date and set it to startdate + one hour.
96            setDateToTextBox(timePicker2, addMilliseconds(time1, ONE_HOUR));
97          }
98          // no need to call onChange for timePicker1, since it will be called
99          // for timePicker2 when changes are made.
100         // listener.onChange(timePicker1);
101         timePicker1PreviousDate = time1;
102       }
103     });
104     timePicker2.addValueChangeHandler(new ValueChangeHandler<String>() {
105       @Override
106       public void onValueChange(ValueChangeEvent<String> valueChangeEvent) {
107         // Hack to stitch together these old and new APIs
108         ChangeEvent event = new ChangeEvent() {
109           @Override
110           public Object getSource() {
111             return timePicker2;
112           }
113         };
114         handler.onChange(event);
115       }
116     });
117   }

119   private void buildTimeZoneList() {
120     for (TimeZoneInfo info : TimeZoneList.TIMEZONES) {
121       timeZones.addItem(info.getGmtRelative() + ' ' + info.getAbbreviation(),
122                         String.valueOf(info.getGmtDiff()));
123     }
124   }

182           element.getStyle().setProperty("display", style);
183         }
184       }

188   }

190   @Override
191   public String getText() throws GeneratorException {
192     String eventName = getEventNameField();
193     String dates = getDateTimeFields();
194     String location = getLocationField();
195     String description = getDescriptionField();
196     StringBuilder output = new StringBuilder();
197     output.append("BEGIN:VEVENT\r\n");
198     output.append(eventName);
199     output.append(dates);
200     output.append(location);
201     output.append(description);
202     output.append("END:VEVENT\r\n");
203     return output.toString();
204   }
205 
206   private String getEventNameField() throws GeneratorException {
207     String inputName = eventName.getText();
208     if (inputName.isEmpty()) {
209       throw new GeneratorException("Event name must be at least 1 character.");
210     }
211     if (inputName.contains("\n")) {
212       throw new GeneratorException("Event name should not contain \\n characters.");
213     }
214     return "SUMMARY:" + inputName + "\r\n";
215   }
216 
217   private String getDateTimeFields() throws GeneratorException {
218     if (fullDay.getValue()) {
219       return getFullDayDateFields();
220     }
221     return getDateTimeValues();
222   }
223 
224   private String getLocationField() throws GeneratorException {
225     String locationString = location.getText();
226     if (locationString.isEmpty()) {
227       return "";
228     }
229     if (locationString.contains("\n")) {
230       throw new GeneratorException(
231           "Location should not contain \\n characters.");
232     }
233     return "LOCATION:" + locationString + "\r\n";
234   }
235 
236   private String getDescriptionField() throws GeneratorException {
237     String descriptionString = description.getText();
238     if (descriptionString.isEmpty()) {
239       return "";
240     }
241     if (descriptionString.contains("\n")) {
242       throw new GeneratorException(
243           "Description should not contain \\n characters.");
244     }
245     return "DESCRIPTION:" + descriptionString + "\r\n";
246   }
247 
248   private String getFullDayDateFields() throws GeneratorException {
249     Date date1 = datePicker1.getValue();
250     Date date2 = datePicker2.getValue();
251     if (date1 == null || date2 == null) {
252       throw new GeneratorException("Start and end dates must be set.");
253     }
254     if (date1.after(date2)) {
255       throw new GeneratorException("End date cannot be before start date.");
256     }
257     // Specify end date as +1 day since it's exclusive
258     Date date2PlusDay = new Date(date2.getTime() + 24 * 60 * 60 * 1000);
259     DateTimeFormat isoFormatter = DateTimeFormat.getFormat("yyyyMMdd");
260     StringBuilder output = new StringBuilder();
261     output.append("DTSTART;VALUE=DATE:");
262     output.append(isoFormatter.format(date1));
263     output.append("\r\n");
264     output.append("DTEND;VALUE=DATE:");
265     output.append(isoFormatter.format(date2PlusDay));
266     output.append("\r\n");
267     return output.toString();
268   }
269 
270   private String getDateTimeValues() throws GeneratorException {
271     Date date1 = datePicker1.getValue();
272     Date date2 = datePicker2.getValue();
273     Date time1 = getDateFromTextBox(timePicker1);
274     Date time2 = getDateFromTextBox(timePicker2);
275     if (date1 == null || date2 == null || time1 == null || time2 == null) {
276       throw new GeneratorException("Start and end dates/times must be set.");
277     }
278     String timezoneDelta = timeZones.getValue(timeZones.getSelectedIndex());
279     long diffTimeZone = Long.parseLong(timezoneDelta);
280     if (summerTime.getValue()) {
281       diffTimeZone += ONE_HOUR;
282     }
283     Date dateTime1 = addMilliseconds(mergeDateAndTime(date1, time1), -diffTimeZone);
284     Date dateTime2 = addMilliseconds(mergeDateAndTime(date2, time2), -diffTimeZone);
285     if (dateTime1.after(dateTime2)) {
286       throw new GeneratorException("Ending date/time cannot be before starting date/time.");
287     }
288     DateTimeFormat isoFormatter = DateTimeFormat.getFormat("yyyyMMdd'T'HHmmss'Z'");
289     StringBuilder output = new StringBuilder();
290     output.append("DTSTART:");
291     output.append(isoFormatter.format(dateTime1));
292     output.append("\r\n");
293     output.append("DTEND:");
294     output.append(isoFormatter.format(dateTime2));
295     output.append("\r\n");
296     return output.toString();
297   }

299   private static Date mergeDateAndTime(Date date, Date time) {
300     // Is that the only ugly way to do with GWT ? given that we don't
301     // have java.util.Calendar for instance
302     DateTimeFormat extractDate = DateTimeFormat.getFormat("yyyyMMdd");
303     DateTimeFormat extractTime = DateTimeFormat.getFormat("HHmm");
304     DateTimeFormat merger = DateTimeFormat.getFormat("yyyyMMddHHmmss");
305     String d = extractDate.format(date);
306     String t = extractTime.format(time) + "00";
307     return merger.parse(d + t);
308   }

310   @Override
311   public void validate(Widget widget) throws GeneratorException {
312     if (widget == eventName) {
313       getEventNameField();
314     } else if (widget == datePicker1 || widget == timePicker1 || widget == datePicker2 || widget == timePicker2) {
315       getDateTimeFields();
316     }
317   }
318 
319   private static Date addMilliseconds(Date time1, long milliseconds) {
320     return new Date(time1.getTime() + milliseconds);
321   }
322 
323   private static Date getDateFromTextBox(HasText textBox) throws GeneratorException {
324     DateTimeFormat extractTime = DateTimeFormat.getFormat("HHmm");
325     try {
326       return extractTime.parseStrict(textBox.getText());
327     } catch (IllegalArgumentException iae) {
328       throw new GeneratorException("Invalid time");
329     }
330   }
331 
332   private static void setDateToTextBox(HasText textBox, Date date) {
333     DateTimeFormat extractTime = DateTimeFormat.getFormat("HHmm");
334     textBox.setText(extractTime.format(date));
335   }
